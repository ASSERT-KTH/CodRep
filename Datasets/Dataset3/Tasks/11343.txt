package org.jboss.as.controller;

import org.jboss.dmr.ModelNode;

/**
 * A model remove operation handler.  A remove operation is an update operation which results in an existing node
 * being removed from the model.  If no node exists at the given address, the operation fails with an error.
 * The submodel is the old content.
 *
 * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
 */
public interface ModelRemoveOperationHandler extends ModelUpdateOperationHandler {

    /**
     * Execute the model remove, passing the result to {@code resultHandler}. This method <b>must</b> invoke one of the
     * completion methods on {@code resultHandler} regardless of the outcome of the operation.
     *
     * @param context the context for this operation
     * @param operation the operation being executed
     * @param resultHandler the result handler to invoke when the operation is complete
     *
     * @return a handle which may be used to asynchronously cancel this operation
     */
    Cancellable execute(NewOperationContext context, ModelNode operation, ResultHandler resultHandler);
}
