     * @return
     *     possible object is
     *     {@link String }
     *

    /**
     * Sets the value of the operationtype property.
     * 
     * @param value
     *     allowed object is
     *     {@link String }

     *     allowed object is

    public void setTextures(Textures value) {

    public Boneassignments getBoneassignments() {
