    public void testNonMoving() {
        Point sp = new Point(10, 10) { public void move(int x, int y) {} };

            sp.move(10, 10);
            fail("should have thrown IllegalStateException");
        } catch (IllegalStateException e) { }

        Point p1 = new Point(10, 100);
        p1.move(3, 30);
    }
