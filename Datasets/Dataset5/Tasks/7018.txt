package org.apache.wicket.examples;

import junit.extensions.TestSetup;
import junit.framework.Test;

import org.mortbay.jetty.Connector;
import org.mortbay.jetty.Server;
import org.mortbay.jetty.nio.SelectChannelConnector;
import org.mortbay.jetty.webapp.WebAppContext;

/**
 * Test decorator that starts a jetty instance
 * 
 * @author ivaynberg
 * 
 */
public class JettyTestCaseDecorator extends TestSetup
{

	private Server server;

	/**
	 * @param test
	 */
	public JettyTestCaseDecorator(Test test)
	{
		super(test);
	}

	protected void setUp() throws Exception
	{
		server = new Server();
		SelectChannelConnector connector = new SelectChannelConnector();
		connector.setPort(8098);
		server.setConnectors(new Connector[] { connector });

		WebAppContext web = new WebAppContext();
		web.setContextPath("/org.apache.wicket-examples");

		String basedir = System.getProperty("basedir");
		String path = "";
		if (basedir != null)
			path = basedir + "/";
		path += "src/main/webapp";

		web.setWar(path);
		server.addHandler(web);

		server.start();

		super.setUp();
	}

	protected void tearDown() throws Exception
	{
		super.tearDown();
		server.stop();
		server.join();
	}

}
