package org.aspectj.weaver.ast;

import org.aspectj.weaver.*;


public class Var extends Expr {
	ResolvedTypeX type;

	public Var(ResolvedTypeX type) {
		super();
		this.type = type;		
	}
    
	public ResolvedTypeX getType() {
		return type;
	}

	public String toString() {
		return "(Var " + type + ")";
	}
    
    public void accept(IExprVisitor v) {
        v.visit(this);
    }
}
