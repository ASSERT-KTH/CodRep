package org.columba.addressbook.folder;

import java.util.*;

/**
 * @version 	1.0
 * @author
 */
public class PropertySet
{
	Vector subkeys;
	Object value;

	public PropertySet()
	{
		subkeys = new Vector();
	}
	
	public void addSubkey( Object key )
	{
		String  s = (String) key;
		
		subkeys.add( s.toLowerCase() );
	}
	
	public void setValue( Object value )
	{
		this.value = value;
	}
	
	public Object getValue()
	{
		return value;
	}
	
	

}
